% This is a hack to make longtable work with twocolumn in pandoc
% From: https://tex.stackexchange.com/a/224096

\makeatletter
\let\oldlt\longtable
\let\endoldlt\endlongtable
\def\longtable{\@ifnextchar[\longtable@i \longtable@ii}
\def\longtable@i[#1]{\begin{figure}[h] % change to figure* for full width
\onecolumn
\begin{minipage}{\textwidth}
\oldlt[#1]
}
\def\longtable@ii{\begin{figure}[h]
\onecolumn
\begin{minipage}{\textwidth}
\oldlt
}
\def\endlongtable{\endoldlt
\end{minipage}
\twocolumn
\end{figure*}}
\makeatother
